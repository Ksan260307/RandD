💡 この仕組みはどう動いている？

このアプリは、毎回の「キャラクターの位置」を保存していません。代わりに、「初期状態」と「ボタンを押した履歴」だけを記録しています。

過去に戻る時や状態を復元する時は、記録された履歴を超高速で再実行して、現在の状態を計算しています。これによりメモリを大幅に節約し、正確な巻き戻しを可能にしています。

// 1. 操作の記録だけを保存
let history = [
  { step: 1, action: 'RIGHT' },
  { step: 2, action: 'LEFT' }
];

// 2. 指定した時間(ステップ)の状態を再計算
function getStateAt(targetStep) {
  let x = 0; // 初期位置
  
  for (let record of history) {
    if (record.step > targetStep) break;
    if (record.action === 'RIGHT') x += 40;
    if (record.action === 'LEFT')  x -= 40;
  }
  
  return x; // 計算された位置を返す
}
